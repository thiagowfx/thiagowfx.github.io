{{- $currentLang := .Site.Language.Lang }} {{- $currentPath := .RelPermalink }} {{- /* Language definitions: code, flag, name, prefix */ -}} {{- $languages := slice (dict "code" "en" "flag" "🇬🇧" "name" "English" "prefix" "") (dict "code" "pt" "flag" "🇧🇷" "name" "Português" "prefix" "/pt") (dict "code" "it" "flag" "🇮🇹" "name" "Italiano" "prefix" "/it") }} {{- /* Determine the current language flag */ -}} {{- $currentLangFlag := "🇬🇧" }} {{- range $languages }} {{- if eq .code $currentLang }} {{- $currentLangFlag = .flag }} {{- end }} {{- end }} {{- /* Strip the current language prefix to get the base path */ -}} {{- $basePath := $currentPath }} {{- range $languages }} {{- if and (ne .prefix "") (strings.HasPrefix $currentPath (printf "%s/" .prefix)) }} {{- $basePath = strings.TrimPrefix .prefix $currentPath }} {{- end }} {{- end }} {{- /* Build paths and check existence for each language */ -}} {{- $availableLangs := slice }} {{- range $lang := $languages }} {{- $path := printf "%s%s" $lang.prefix $basePath }} {{- if eq $basePath "/" }} {{- if eq $lang.prefix "" }} {{- $path = "/" }} {{- else }} {{- $path = printf "%s/" $lang.prefix }} {{- end }} {{- end }} {{- $exists := false }} {{- /* Root path always exists for the default language */ -}} {{- if and (eq $path "/") (eq $lang.code "en") }} {{- $exists = true }} {{- else if eq $path (printf "%s/" $lang.prefix) }} {{- /* Language root without a specific page — skip */ -}} {{- else }} {{- range hugo.Sites }} {{- if eq .Language.Lang $lang.code }} {{- range .Pages }} {{- if eq .RelPermalink $path }} {{- $exists = true }} {{- end }} {{- end }} {{- end }} {{- end }} {{- end }} {{- if $exists }} {{- $availableLangs = $availableLangs | append (dict "code" $lang.code "flag" $lang.flag "name" $lang.name "path" $path) }} {{- end }} {{- end }} {{- /* Only show language switcher if more than one language exists */ -}} {{- if gt (len $availableLangs) 1 }}
{{- range $availableLangs }} {{ .flag }} {{ .name }} {{- end }}
{{- end }}
